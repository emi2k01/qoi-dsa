use crate::{ decoder::ChunksDecoder, error::DecodeImageError, header::{Channels, Header}, pixel::Pixel, }; pub struct Image { header: Header, pixels: Vec<Pixel>, } impl Image { pub fn from_slice(input: &[u8]) -> Result<Self, DecodeImageError> { let header_input = input[0..14] .try_into() .map_err(|_| DecodeImageError::InvalidLength(input.len()))?; let header = Header::decode(header_input)?; let size = header.width * header.height; let chunks_input = input[14..].iter().copied(); let pixels = ChunksDecoder::from_iter(chunks_input).decode(size)?; Ok(Self { header, pixels }) } fn to_raw_bitmap(&self) -> Vec<u8> { let pixels = match self.header.channels { Channels::Rgb => self .pixels .iter() .copied() .flat_map(|p| [p.r, p.g, p.b]) .collect(), Channels::Rgba => self .pixels .iter() .copied() .flat_map(|p| [p.r, p.g, p.b, p.a]) .collect(), }; pixels } } #[cfg(test)] mod tests { use super::*; macro_rules!
